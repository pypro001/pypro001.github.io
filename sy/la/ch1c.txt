<code>
import matplotlib.pyplot as plt

numb = int(input("enter no. of points to be plotted : "))
x = []
y = []
for jack in range(0, numb):
    re = float(input("enter the real part: "))
    im = float(input('enter the imaginary part: '))
    x.append(re)
    y.append(im)
plt.scatter(x, y, label="star", marker="*", color="green", s=30)
plt.xlabel('real axis')
plt.ylabel('imaginary axis')
plt.title('complex numbers')
plt.legend()
plt.show()
