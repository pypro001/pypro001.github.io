4a.While-loop Statements._______

DECLARE
i INTEGER := 1;
BEGIN
WHILE i <= 10 LOOP
DBMS_OUTPUT.PUT_LINE(i);
i := i+1;
END LOOP;
END;

____________

4b.for loop statement__________

DECLARE
 a number(2);
BEGIN
FOR a in 10 .. 20 LOOP
dbms_output.put_line('value of a: '||a);
END LOOP;
END;
