package org.me.conversion;
import javax.jws.WebService;
import javax.jws.WebMethod;
import javax.jws.WebParam;
@WebService(serviceName = "conversion")
public class conversion {
 @WebMethod(operationName = "fahrenheittocelciues")
 public double temptofarh(@WebParam(name = "F") double F) {
 double c=((F-32)*5)/9;
 return c; 
 }
 @WebMethod(operationName = "celciuestofahrenheit")
 public double farhtotemp(@WebParam(name = "C") double C) {
 double f=(C*9/5)+32;
 return f;
 }
}
