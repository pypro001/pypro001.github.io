@WebService(serviceName = "Operation")

 public int add(@WebParam(name = "n1") int n1, @WebParam(name = "n2") int n2) {
 int c=n1+n2;
 return c;
 }

 
 @WebMethod(operationName = "sub")
 public int sub(@WebParam(name = "n1") int n1, @WebParam(name = "n2") int n2) {
 int c=n1-n2;
 return c;
 }
 @WebMethod(operationName = "Mul")
 public int Mul(@WebParam(name = "n1") int n1, @WebParam(name = "n2") int n2) {
 int c=n1*n2;
 return c;
 }
 @WebMethod(operationName = "div")
 public int div(@WebParam(name = "n1") int n1, @WebParam(name = "n2") int n2) {
 int c=n1/n2;
 return c;
 }
}
