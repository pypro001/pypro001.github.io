package railfencecipher; 
import java.util.Scanner; 
public class RailFenceCipher
{
public String Encrypt(String pt)
{
String even="",odd=""; if((pt.length())%2!=0)
{
pt+="x";
}
for(int i=0;i<pt.length();i++)
{
if(i%2==0)
{
even+=pt.charAt(i);
}
else
{
odd+=pt.charAt(i);
}
}
return(even+odd);
}
public String Decrypt(String ct)
{
String pt="",s1="",s2=""; 
s1=ct.substring(0,(ct.length())/2); 
s2=ct.substring((ct.length())/2); 
for(int i=0;i<ct.length()/2;i++)
{
pt+=s1.charAt(i)+""+s2.charAt(i);
}
return pt;
}
public static void main(String[] args) { 
    System.out.println("Rail Fence Cipher"); 
    System.out.println("Enter the plaintext: "); 
    Scanner sc=new Scanner(System.in); 
    String pt=sc.next();
    RailFenceCipher r=new RailFenceCipher(); 
    String ct=r.Encrypt(pt); 
    System.out.println("Encrytion: "+r.Encrypt(pt)); 
    System.out.println("Decryption: "+r.Decrypt(ct));
}
}

package columnarcipher; 
import java.util.Scanner; 
public class ColumnarCipher {
public String Encryption(String pt,int col)
{
char matrix[][]=new char[5][5]; 
int cursor=0; 
pt=pt.toUpperCase(); 
if(pt.length()%2!=0)
{ pt+= ' ';
}
for(int i=0;i<pt.length()/col;i++)
{
for(int j=0;j<col;j++)
{
matrix[i][j]=pt.charAt(cursor); 
System.out.print(matrix[i][j]+" "); 
cursor++;
}
System.out.println();
}
System.out.println("Encrption:"); 
for(int i=0;i<col;i++)
{
for(int j=0;j<pt.length()/col;j++)
{
System.out.print(matrix[j][i]+" ");
}
}
return pt;
}
public String Decryption(String ct,int col)
{
ct=ct.toLowerCase(); 
System.out.println("\nDecryption:"); 
char matrix[][]=new char[5][5]; 
for(int i=0;i<ct.length()/col;i++)
{
for(int j=0;j<col;j++)
{
System.out.print(matrix[i][j]);
}
}
return ct;
}

    System.out.println("Enter the plaintext: ");

System.out.println("Enter the column number: "); 
int col=sc.nextInt();
