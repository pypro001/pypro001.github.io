class FSM:
    def _init_(self):
        self.current_state = 'q0'

    def transition(self, input):
        if self.current_state == 'q0' and input == '1':
            self.current_state = 'q1'
        elif self.current_state == 'q1' and input == '0':
            self.current_state = 'q2'
        elif self.current_state == 'q2' and input == '1':
            self.current_state = 'q3'
            print("Accepted")
        elif input == '0':
            self.current_state = 'q0'
        else:
            self.current_state = 'q1'

# create FSM object
fsm = FSM()

# test input strings
input_strings = ['001101', '101101', '11101101', '1111101']

# feed input strings to FSM
for string in input_strings:
    for char in string:
        fsm.transition(char)

#include <iostream>
#include <string>

using namespace std;

// DFA states
enum State {
    START,
    ONE,
    TEN,
    HUNDRED,
    ACCEPT
};

// Check if the input string is accepted by the DFA
bool accept(string input) {
    State state = START;

    for (char c : input) {
        switch (state) {
            case START:
                if (c == '1') {
                    state = ONE;
                }
                break;
            case ONE:
                if (c == '0') {
                    state = TEN;
                } else if (c != '1') {
                    state = START;
                }
                break;
            case TEN:
                if (c == '1') {
                    state = HUNDRED;
                } else if (c != '0') {
                    state = START;
                }
                break;
            case HUNDRED:
                if (c == '0') {
                    state = TEN;
                } else if (c != '1') {
                    state = START;
                }
                break;
            case ACCEPT:
                return false; // Input is longer than "101"
        }
    }

    return state == HUNDRED;
}

int main() {
    string input;
    cout << "Enter an input string: ";
    cin >> input;

    if (accept(input)) {
        cout << "Input accepted!" << endl;
    } else {
        cout << "Input not accepted." << endl;
    }

    return 0;
}
