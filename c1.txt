fun main() {
    // if-else statement
    val x = 10
    if (x > 5) {
        println("x is greater than 5")
    } else {
        println("x is less than or equal to 5")
    }

    // when expression
    val dayOfWeek = "Wednesday"
    val message = when (dayOfWeek) {
        "Monday" -> "Back to work"
        "Friday" -> "Thank goodness it's Friday"
        else -> "Another day"
    }
    println(message)

    // for loop
    for (i in 1..5) {
        println("Counting: $i")
    }

    // while loop
    var count = 1
    while (count <= 5) {
        println("While loop count: $count")
        count++
    }

    // do-while loop
    var number = 0
    do {
        println("Do-while loop number: $number")
        number++
    } while (number < 5)

    // continue and break statements
    for (i in 1..10) {
        if (i % 2 == 0) {
            continue // skip even numbers
        }
        if (i == 7) {
            break // exit loop when i equals 7
        }
        println(i)
    }
}
